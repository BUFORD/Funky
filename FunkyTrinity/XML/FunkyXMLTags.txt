AddProfileSet
	<AddProfileSet Profiles="A.xml,B.xml,C.xml" Random="true" />
		--Adds a set of profiles that can be randomized. A set is actively used until it is emptied by using the <NextProfile /> tag.
	Exmaple:
		    <AddProfileSet profiles="A2_Blackmines.xml,A2_HowlingPlateu_1.xml" random="false" />
			<AddProfileSet profiles="A2_DesolateSands.xml,A2_Oasis.xml,A2_Alcarnus.xml" random="true" />

CheckScenes
	<CheckScenes Scenes="0000,0001,0002" />
		--Using this will set an ID if any of the given scene IDs are currently present. To check which scene ID was set, see <Scene /> Tag.

FunkyCastTP
	<FunkyCastTP />
		--Uses custom behavior to return to town by casting TP. It also clears the area prior to attempting!

NextProfile
	<NextProfile />
		--To be used in conjuction with the AddProfileSet tag!
		--No profiles remaining will attempt to find the start profile, which searching for profiles that contain "start" in the name.

Scene
	Example Use with <CheckScenes> tag:

	<CheckScenes Scenes="31604,31603,72696" />
    <Scene id="31604">
		<TrinityUseReset questId="1" id="31604" />
    </Scene>


SetProfileVar
	<SetProfileVar startProfile="A.xml" lastProfile="Z.xml" />
		--It is not necessary to set either of these, but if using NextProfile than it is recommended to set at least the starting profile!

	Example:
	<SetProfileVar startProfile="START_A2_Only.xml" lastProfile="A2_ZoltunKulle.xml" />
